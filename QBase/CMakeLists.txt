CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

PROJECT(BERTLIB)

SET(CMAKE_CXX_FLAGS_DEBUG   "-g -Wall")
#SET(CMAKE_CXX_FLAGS_DEBUG   "-g -Wall -DDEBUG_BERT_SDK")

SET(CMAKE_BUILD_TYPE "Debug")

AUX_SOURCE_DIRECTORY(. SDK_SRC)
AUX_SOURCE_DIRECTORY(./Log     SDK_SRC)
AUX_SOURCE_DIRECTORY(./Threads SDK_SRC)
AUX_SOURCE_DIRECTORY(./SmartPtr SDK_SRC)
SET(LIBRARY_OUTPUT_PATH ../)
ADD_LIBRARY(bertlib STATIC ${SDK_SRC})

TARGET_LINK_LIBRARIES(bertlib pthread)

SET_TARGET_PROPERTIES(bertlib PROPERTIES LINKER_LANGUAGE CXX)
